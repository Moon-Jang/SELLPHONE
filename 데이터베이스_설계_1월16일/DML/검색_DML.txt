--검색 '갤럭시'--
SELECT PRODUCT_NAME,PRODUCT_PRICE
FROM PRODUCT
WHERE PRODUCT_NAME LIKE '%갤럭시%'

--검색 '아이폰' AND 가격 50~100만 사이--
SELECT PRODUCT_NAME,PRODUCT_PRICE
FROM PRODUCT
WHERE PRODUCT_NAME LIKE '%아이폰%'
AND (PRODUCT_PRICE BETWEEN 500000 AND 1000000);

--검색 '갤럭시노트' 가격 10~30만 사이--
SELECT PRODUCT_NAME,PRODUCT_PRICE
FROM PRODUCT
WHERE PRODUCT_NAME LIKE '%갤럭시노트%'
AND (PRODUCT_PRICE BETWEEN 100000 AND 1000000);
